# Basic KPIs
Total Transactions =
COUNT ( 'synthetic_fraud_dataset'[transaction_id] )


Total Fraud Transactions =
CALCULATE (
    COUNT ( 'synthetic_fraud_dataset'[transaction_id] ),
    'synthetic_fraud_dataset'[is_fraud] = 1
)


Total Legit Transactions =
CALCULATE (
    COUNT ( 'synthetic_fraud_dataset'[transaction_id] ),
    'synthetic_fraud_dataset'[is_fraud] = 0
)


-- ğŸ”¹ Amount-Based KPIs
Total Transaction Amount =
SUM ( 'synthetic_fraud_dataset'[amount] )


Total Fraud Amount =
CALCULATE (
    SUM ( 'synthetic_fraud_dataset'[amount] ),
    'synthetic_fraud_dataset'[is_fraud] = 1
)


Total Legit Amount =
CALCULATE (
    SUM ( 'synthetic_fraud_dataset'[amount] ),
    'synthetic_fraud_dataset'[is_fraud] = 0
)


-- ğŸ”¹ Identification Metrics
Fraud %
=
DIVIDE (
    [Total Fraud Transactions],
    [Total Transactions],
    0
)


Fraud Amount %
=
DIVIDE (
    [Total Fraud Amount],
    [Total Transaction Amount],
    0
)


-- ğŸ”¹ Risk-Related Calculations
Avg Device Risk Score =
AVERAGE ( 'synthetic_fraud_dataset'[device_risk_score] )


Avg IP Risk Score =
AVERAGE ( 'synthetic_fraud_dataset'[ip_risk_score] )


Fraud Score Index =
DIVIDE (
    [Avg Device Risk Score] + [Avg IP Risk Score],
    2,
    0
)


-- ğŸ”¹ Temporal Insights (Hour-based detection)
Fraud Transactions (Night)
=
CALCULATE (
    [Total Fraud Transactions],
    'synthetic_fraud_dataset'[hour] >= 0
        && 'synthetic_fraud_dataset'[hour] <= 6
)


Fraud Transactions (Day)
=
CALCULATE (
    [Total Fraud Transactions],
    'synthetic_fraud_dataset'[hour] >= 7
        && 'synthetic_fraud_dataset'[hour] <= 19
)
